\section{ISDN}
\subsection{Definition und Merkmale}
\begin{frame}
\begin{itemize}
\item Integrated Services Digital Network
\item Internationaler Standard für digitales Kommunikationsnetzwerk
\item Integration verschiedener Dienste z.B. Telefonie, Telefax, Datenübertragung
\item Zwei Anschlussvarianten
\begin{itemize}
  \item Basisanschluss
  \item Anlagenanschluss
\end{itemize}
%  \item fictitious settings \& actions
%  \vskip0pt plus.3fill
%  \item scripted or based on spontaneous acting
%  \vskip0pt plus.3fill
%  \item being someone else than yourself
\end{itemize}
\end{frame}

\subsubsection*{Basisanschluss}
\begin{frame}
\frametitle{Basisanschluss}
\begin{itemize}
  \item Zwei Nutzkanäle (B-Kanäle) mit je 64 Kbit/s
  \begin{itemize}
    \item Telefonie, Telefax \ldots
  \end{itemize}
  \item Ein Steuerkanal (D-Kanal) mit 16 Kbit/s
  \begin{itemize}
    \item Steuerinformationen
  \end{itemize}
  \item Nutzung von Privatkunden und kleineren Betrieben
\end{itemize}
\end{frame}

\subsubsection*{Anlagenanschluss}
\begin{frame}
\frametitle{Anlagenanschluss}
\begin{itemize}
  \item Anschluss einer Telefonanlage
  \item Eine Grundrufnummer mit beliebiger Anzahl von Durchwahlen
  \item Nutzung von mittelständischen bis großen Betrieben 
\end{itemize}
\end{frame}

\subsubsection*{Primärmultiplex-Anschluss}
\begin{frame}
\frametitle{Primärmultiplex-Anschluss}
\begin{itemize}
  \item Besondere Form des Anlagenanschlusses
  \item 30 B-Kanäle mit je 64 Kbit/s
  \item Ein D-Kanal mit 64 Kbit/s
  \item Ein Kanal für Synchronisation und Wartung mit 64 Kbit/s
\end{itemize}
\end{frame}

\subsection{Vergleich von ISDN- und Analog-Anschlüssen}
\begin{frame}
\begin{small}
\begin{columns}[c]
\column{1.8in}
Analog-Anschlüsse:
	\begin{itemize}
      \item 1 Kanal
      \item 1 Rufnummer
      \item 56 Kbit/s
      \item Signalverstärkung\\
      \item Basispreis: 17~\euro/Monat
    \end{itemize}
\column{1.8in}
ISDN-Anschlüsse:
	\begin{itemize}
      \item 2 Kanäle
      \item 1 - 10 Rufnummern
      \item 128 Kbit/s
      \item Signalregenerierung\\
      \item Basispreis: 17,90~\euro/Monat
    \end{itemize}
\end{columns}
\end{small}
\end{frame}

\section{DSL}
\subsection{Definition und Merkmale}
\begin{frame}
\frametitle{DSL}
	\begin{itemize}
      \item Digital Subscriber Line
      \item Übertragungsstandard mit bis zu 200 MBit via Kupferkabel
      \item DSL ``standalone''
      \item DSL über POTS
      \item DSL über ISDN
    \end{itemize}
    % FREQUENZSPEKTRUM-BILD!
\end{frame}

\subsection{ADSL}
\begin{frame}
\frametitle{ADSL}
	\begin{itemize}
      \item Asymmetric Digital Subscriber Line
      \item Primär für Privatanwender
      \item Downstream höher als Upstream
      \item ADSL
      \item ADSL2 \& ADSL2+
    \end{itemize}
    % T-DSL-Angebot
\end{frame}

\subsection{SDSL}
\begin{frame}
\frametitle{SDSL}
	\begin{itemize}
      \item Symmetric Digital Subscriber Line
      \item Primär für Businessanwender
      \item Downstream entspricht Upstream
      \item Kein Festnetzanschluss möglich, deswegen als Datenanschluss bezeichnet
      \item Aktuell bis zu 20 MBit realisierbar
    \end{itemize}
\end{frame}
